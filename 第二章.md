function student(){
    this.mark = [];
	this.add = add;
	this.average = average;
}

function add(temp){
    this.mark.push(temp);
}

function average(){
	var total = 0;
	for(var i= 0; i<this.mark.length; ++i){
		total +=this.mark[i];
    }
	return total/this.mark.length;
}
var st = new student();
st.add(80);
st.add(81);
st.add(82);
console.log(st.average());
